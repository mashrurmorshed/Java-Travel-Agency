create table account(
ID integer primary key autoincrement,
Name char not null,
Email char not null unique,
Phone char,
Username char not null unique,
Password char not null,
Trips integer default 0
);

create table hotel(
ID integer primary key autoincrement,
Location char not null,
Name char not null
);

create table room(
ID integer primary key autoincrement,
hotel_id integer,
roomno char,
cost integer,
foreign key(hotel_id) references hotel(ID)
);

create table bus_transport(
ID integer primary key autoincrement,
Location char not null,
Name char not null
);

create table bus_seats(
ID integer primary key autoincrement,
bus_id integer,
seatno char,
cost integer,
foreign key(bus_id) references bus_transport(ID)
);

create table air_transport(
ID integer primary key autoincrement,
Location char not null,
Name char not null
);

create table air_seats(
ID integer primary key autoincrement,
air_id integer,
seatno char,
cost integer,
foreign key(air_id) references air_transport(ID)
);

create table rate(
ID integer primary key autoincrement,
type integer,
rating integer
);
